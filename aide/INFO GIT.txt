AVANT
# 1. Récupérer les dernières modifications de l'équipe
git pull origin main

# 2. Créer une branche pour ta fonctionnalité du jour
git checkout -b feat/nom-de-la-feature

# Exemples de bons noms de branches :
# feat/ingredient-model
# feat/meal-log-crud
# fix/calcul-macros

PENDANT
# Vérifier régulièrement ce que tu as modifié
git status

# Voir le détail des modifications
git diff

FIN 
# 1. Ajouter tes fichiers modifiés
git add .

# 2. Commiter avec un message clair
git commit -m "feat: ajout modèle SQLAlchemy Ingredient"

# 3. Pousser ta branche sur GitHub
git push origin feat/nom-de-la-feature
